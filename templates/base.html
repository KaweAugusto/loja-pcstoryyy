from django.db import models from cloudinary.models import CloudinaryField class Categoria(models.Model): nome = models.CharField(max_length=100, unique=True) descricao = models.TextField(blank=True, null=True) class Meta: verbose_name = "Categoria" verbose_name_plural = "Categorias" def __str__(self): return self.nome class Produto(models.Model): nome = models.CharField(max_length=200) descricao = models.TextField(blank=True, null=True) preco = models.DecimalField(max_digits=10, decimal_places=2) estoque = models.PositiveIntegerField(default=0) categoria = models.ForeignKey( Categoria, on_delete=models.CASCADE, related_name='produtos' # <-- ALTERAÇÃO 1 ) imagem = CloudinaryField('image', blank=True, null=True) criado_em = models.DateTimeField(auto_now_add=True) atualizado_em = models.DateTimeField(auto_now=True) class Meta: verbose_name = "produto" verbose_name_plural = "produtos" # <-- ALTERAÇÃO 2 def __str__(self): return self.nome
